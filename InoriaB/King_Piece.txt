                            TextLabel.TextScaled = true
                            end
                        end
                    end
                end
            end

            _G.KINGdevilstuff = false

                if v.ClassName == "BillboardGui" and v.Parent.Parent:IsA("Tool") and v.Parent.Parent:FindFirstChild("DevilFruit") then
